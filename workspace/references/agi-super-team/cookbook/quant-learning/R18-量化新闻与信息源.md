### 3.2 X/Twitter 数据

```python
# Twitter/X 是加密市场信息最快来源

# 方式1: 使用 x_search 工具（推荐）
# x_search(query="bitcoin", from_date="2024-01-01")

# 方式2: Tweepy (需要X API)
# pip install tweepy
import tweepy

class TwitterSentiment:
    def __init__(self, bearer_token):
        self.client = tweepy.Client(bearer_token=bearer_token)

    def search_crypto_tweets(self, query='Bitcoin', max_results=100):
        """搜索加密相关推文"""
        tweets = self.client.search_recent_tweets(
            query=f'{query} -is:retweet lang:en',
            max_results=max_results,
            tweet_fields=['created_at', 'public_metrics', 'sentiment']
        )
        return tweets.data

    def get_sentiment(self, tweets):
        """简单情感分析"""
        from textblob import TextBlob

        sentiments = []
        for tweet in tweets:
            blob = TextBlob(tweet.text)
            sentiments.append({
                'text': tweet.text[:100],
                'polarity': blob.sentiment.polarity,  # -1到1
                'subjectivity': blob.sentiment.subjectivity,  # 0到1
            })

        df = pd.DataFrame(sentiments)
        return {
            'mean_polarity': df['polarity'].mean(),
            'positive_ratio': (df['polarity'] > 0).mean(),
            'negative_ratio': (df['polarity'] < 0).mean(),
        }
```

## 四、NLP情感分析

### 4.1 金融情感分析

```python
# pip install transformers torch
from transformers import pipeline

class FinancialSentimentAnalyzer:
    """金融文本情感分析"""

    def __init__(self, model_name='ProsusAI/finbert'):
        """
        FinBERT: 金融领域预训练BERT
        """
        self.classifier = pipeline(
            'sentiment-analysis',
            model=model_name,
        )

    def analyze(self, texts):
        """
        分析文本情感
        返回: positive, negative, neutral 概率
        """
        results = []
        for text in texts:
            result = self.classifier(text)[0]
            results.append({
                'text': text[:100],
                'label': result['label'],
                'score': result['score'],
            })
        return pd.DataFrame(results)

    def aggregate_sentiment(self, results_df):
        """聚合情感得分"""
        sentiment_score = (
            (results_df['label'] == 'positive').sum() -
            (results_df['label'] == 'negative').sum()
        ) / len(results_df)
        return sentiment_score

# 使用
analyzer = FinancialSentimentAnalyzer()
news = ["Bitcoin surges past $70,000", "SEC rejects ETF application"]
results = analyzer.analyze(news)
print(results)
```

```python
class NewsDrivenStrategy:
    """新闻事件驱动策略"""

    def __init__(self, sentiment_threshold=0.3):
        self.threshold = sentiment_threshold
        self.sentiment_history = []

    def process_news(self, news_items):
        """处理新闻，生成信号"""
        # 分析情感
        sentiments = []
        for item in news_items:
            # 简化：用关键词匹配
            score = self._keyword_sentiment(item['title'])
            sentiments.append(score)

        # 聚合
        avg_sentiment = np.mean(sentiments)
        self.sentiment_history.append(avg_sentiment)

        # 生成信号
        if avg_sentiment > self.threshold:
            return 1   # 正面 → 做多
        elif avg_sentiment < -self.threshold:
            return -1  # 负面 → 做空
        return 0

    def _keyword_sentiment(self, text):
        """关键词情感"""
        positive_words = ['surge', 'rally', 'bullish', 'breakthrough', 'adoption',
                         '涨', '利好', '突破', '新高']
        negative_words = ['crash', 'bearish', 'ban', 'reject', 'hack',
                         '跌', '利空', '暴跌', '崩盘']

        text_lower = text.lower()
        score = 0
        for w in positive_words:
            if w in text_lower:
                score += 1
        for w in negative_words:
            if w in text_lower:
                score -= 1
        return score
```
