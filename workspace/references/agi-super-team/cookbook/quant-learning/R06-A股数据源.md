## 四、数据获取最佳实践

### 4.1 数据下载脚本模板

```python
"""
A股量化数据下载器
自动下载并缓存股票数据
"""
import akshare as ak
import pandas as pd
import os
from datetime import datetime, timedelta
import time

class AShareDataDownloader:
    def __init__(self, cache_dir='./data'):
        self.cache_dir = cache_dir
        os.makedirs(cache_dir, exist_ok=True)

    def get_stock_list(self):
        """获取全部A股列表"""
        cache_file = f'{self.cache_dir}/stock_list.csv'
        if os.path.exists(cache_file):
            age = (datetime.now() - datetime.fromtimestamp(
                os.path.getmtime(cache_file))).days
            if age < 7:  # 7天内的缓存
                return pd.read_csv(cache_file)

        df = ak.stock_zh_a_spot_em()
        df.to_csv(cache_file, index=False)
        time.sleep(1)
        return df

    def get_daily_data(self, symbol, start_date='20200101',
                       end_date=None, force=False):
        """获取日线数据"""
        if end_date is None:
            end_date = datetime.now().strftime('%Y%m%d')

        cache_file = f'{self.cache_dir}/daily/{symbol}.parquet'

        if not force and os.path.exists(cache_file):
            df = pd.read_parquet(cache_file)
            last_date = df['日期'].max()
            if last_date >= end_date:
                return df
            # 增量更新
            new_data = ak.stock_zh_a_hist(
                symbol=symbol, period='daily',
                start_date=str(last_date), end_date=end_date,
                adjust='qfq'
            )
            df = pd.concat([df, new_data]).drop_duplicates()
            time.sleep(0.5)
        else:
            df = ak.stock_zh_a_hist(
                symbol=symbol, period='daily',
                start_date=start_date, end_date=end_date,
                adjust='qfq'
            )
            time.sleep(0.5)

        os.makedirs(os.path.dirname(cache_file), exist_ok=True)
        df.to_parquet(cache_file, index=False)
        return df

    def download_index_constituents(self, index_code='000300'):
        """下载指数成分股"""
        df = ak.index_stock_cons_csindex(symbol=index_code)
        return df
```

### 4.2 数据质量检查

```python
def check_data_quality(df):
    """检查数据质量"""
    issues = []

    # 1. 缺失值检查
    missing = df.isnull().sum()
    if missing.any():
        issues.append(f"缺失值: {missing[missing > 0].to_dict()}")

    # 2. 异常值检查
    if 'close' in df.columns:
        daily_return = df['close'].pct_change()
        extreme = daily_return[daily_return.abs() > 0.2]
        if len(extreme) > 0:
            issues.append(f"极端收益(>20%): {len(extreme)}天")

    # 3. 交易量异常
    if 'volume' in df.columns:
        zero_vol = (df['volume'] == 0).sum()
        if zero_vol > 0:
            issues.append(f"零成交量: {zero_vol}天")

    # 4. 连续停牌
    if 'close' in df.columns:
        no_change = (df['close'].diff() == 0).sum()
        if no_change > len(df) * 0.05:
            issues.append(f"价格无变化天数过多: {no_change}")

    return issues if issues else ["数据质量良好 ✅"]
```
