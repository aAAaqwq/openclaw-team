## 二、指数增强策略

### 2.1 核心逻辑

在跟踪指数的基础上，通过因子暴露或个股权重调整获取超额收益。

**目标：** 跑赢基准指数（如沪深300、中证500），同时控制跟踪误差。

### 2.2 实现方法

```python
import numpy as np
import pandas as pd

class IndexEnhancement:
    """沪深300指数增强策略"""

    def __init__(self, benchmark='000300'):
        self.benchmark = benchmark

    def optimize_weights(self, factor_scores, benchmark_weights, 
                         max_tracking_error=0.05, max_stock_weight=0.05):
        """
        基于因子得分优化权重
        factor_scores: Series (stock → score)
        benchmark_weights: Series (stock → weight)
        """
        # 基准权重 + 因子偏离
        enhanced = benchmark_weights.copy()

        # 按因子得分调整权重
        score_rank = factor_scores.rank(pct=True)
        for stock in enhanced.index:
            if stock in score_rank.index:
                # 高分股票增加权重，低分减少
                deviation = (score_rank[stock] - 0.5) * 0.02  # 最大偏离2%
                enhanced[stock] = min(
                    enhanced[stock] + deviation,
                    max_stock_weight
                )

        # 归一化
        enhanced = enhanced / enhanced.sum()
        return enhanced

    def calculate_tracking_error(self, portfolio_returns, benchmark_returns):
        """计算跟踪误差"""
        excess = portfolio_returns - benchmark_returns
        return excess.std() * np.sqrt(252)
```

## 三、统计套利

### 3.1 核心逻辑

利用统计规律发现价格偏离，做多低估、做空高估的资产组合。

### 3.2 配对交易（Pairs Trading）

```python
import numpy as np
import pandas as pd
from statsmodels.tsa.stattools import coint

class PairsTrading:
    """配对交易策略"""

    def find_cointegrated_pairs(self, price_df, significance=0.05):
        """
        寻找协整配对
        price_df: DataFrame (date × stock), 收盘价
        """
        stocks = price_df.columns
        n = len(stocks)
        pairs = []

        # 计算协整检验矩阵
        for i in range(n):
            for j in range(i+1, n):
                S1 = price_df[stocks[i]].dropna()
                S2 = price_df[stocks[j]].dropna()

                common = S1.index.intersection(S2.index)
                if len(common) < 100:
                    continue

                score, pvalue, _ = coint(S1[common], S2[common])
                if pvalue < significance:
                    # 计算价差 (spread)
                    spread = S1[common] / S2[common]
                    zscore = (spread - spread.mean()) / spread.std()

                    pairs.append({
                        'stock1': stocks[i],
                        'stock2': stocks[j],
                        'pvalue': pvalue,
                        'spread_mean': spread.mean(),
                        'spread_std': spread.std(),
                        'current_zscore': zscore.iloc[-1]
                    })

        return pd.DataFrame(pairs).sort_values('pvalue')

    def generate_signals(self, spread, entry_z=2.0, exit_z=0.5):
        """
        生成交易信号
        spread: 价差序列
        entry_z: 开仓阈值
        exit_z: 平仓阈值
        """
        zscore = (spread - spread.rolling(60).mean()) / spread.rolling(60).std()

        signals = pd.Series(0, index=spread.index)
        position = 0

        for i in range(1, len(zscore)):
            if position == 0:  # 无仓位
                if zscore.iloc[i] > entry_z:     # 价差过高
                    signals.iloc[i] = -1          # 做空spread
                    position = -1
                elif zscore.iloc[i] < -entry_z:   # 价差过低
                    signals.iloc[i] = 1           # 做多spread
                    position = 1

            elif position == 1:  # 做多spread
                if zscore.iloc[i] > -exit_z:      # 回归到均值附近
                    signals.iloc[i] = 0           # 平仓
                    position = 0

            elif position == -1:  # 做空spread
                if zscore.iloc[i] < exit_z:       # 回归到均值附近
                    signals.iloc[i] = 0           # 平仓
                    position = 0

        return signals
```

**A股配对交易要点：**
- 同行业/同板块股票更适合配对（如银行股对）
- ⚠️ A股做空困难（融券成本高、券源少）
- 实战中更常用"多空都做多，但比例不同"的方式

## 四、趋势跟踪策略

### 4.1 双均线交叉

```python
class DualMACross:
    """双均线交叉策略"""

    def __init__(self, short_window=5, long_window=20):
        self.short = short_window
        self.long = long_window

    def generate_signals(self, prices):
        """生成信号"""
        signals = pd.Series(0, index=prices.index)

        ma_short = prices.rolling(self.short).mean()
        ma_long = prices.rolling(self.long).mean()

        # 金叉做多，死叉做空
        signals[ma_short > ma_long] = 1
        signals[ma_short < ma_long] = -1

        return signals, ma_short, ma_long
```

### 4.2 海龟交易法（A股版）

```python
class TurtleTrading:
    """海龟交易法 — A股适配版"""

    def __init__(self, entry_window=20, exit_window=10, atr_window=20):
        self.entry_window = entry_window
        self.exit_window = exit_window
        self.atr_window = atr_window

    def calculate_atr(self, high, low, close):
        """计算ATR"""
        tr = pd.DataFrame({
            'hl': high - low,
            'hc': (high - close.shift(1)).abs(),
            'lc': (low - close.shift(1)).abs()
        }).max(axis=1)
        return tr.rolling(self.atr_window).mean()

    def generate_signals(self, df):
        """
        df: 包含 open, high, low, close 列
        """
        df['atr'] = self.calculate_atr(df['high'], df['low'], df['close'])
        df['entry_high'] = df['high'].rolling(self.entry_window).max()
        df['entry_low'] = df['low'].rolling(self.entry_window).min()
        df['exit_high'] = df['high'].rolling(self.exit_window).max()
        df['exit_low'] = df['low'].rolling(self.exit_window).min()

        signals = pd.Series(0, index=df.index)
        position = 0

        for i in range(self.entry_window, len(df)):
            if position == 0:
                if df['close'].iloc[i] > df['entry_high'].iloc[i-1]:
                    signals.iloc[i] = 1  # 突破买入
                    position = 1
            elif position == 1:
                if df['close'].iloc[i] < df['exit_low'].iloc[i-1]:
                    signals.iloc[i] = 0  # 止盈/止损
                    position = 0

        return signals
```

## 五、日内回转策略

### 5.1 概述

日内回转（Day Trading）指在同一交易日内完成买卖，利用日内价格波动获利。A股T+1规则下，日内回转有以下方式：

| 方式 | 说明 | 门槛 |
|------|------|------|
| **底仓T+0** | 利用已有底仓做日内高抛低吸 | 需要底仓 |
| **融券T+0** | 融券卖出 → 买券还券 | 需要融券资格 |
| **ETF T+0** | 买卖跨境ETF/债券ETF | 无特殊门槛 |
| **可转债T+0** | 交易可转债 | 无特殊门槛 |

### 5.2 底仓日内回转示例

```python
class DayTrading:
    """日内回转策略（基于底仓）"""

    def __init__(self, holding_shares=10000):
        self.holding = holding_shares  # 底仓数量
        self.max_trade = 5000          # 单次最大交易量

    def generate_signals(self, minute_data):
        """
        minute_data: 分钟级数据 (time, open, high, low, close, volume)
        返回: 交易信号列表
        """
        signals = []
        position = 0  # 日内仓位（正=多，负=空底仓）

        # 计算VWAP
        minute_data['cum_vol'] = minute_data['volume'].cumsum()
        minute_data['cum_amount'] = (
            minute_data['close'] * minute_data['volume']
        ).cumsum()
        minute_data['vwap'] = (
            minute_data['cum_amount'] / minute_data['cum_vol']
        )

        # 计算布林带
        minute_data['ma20'] = minute_data['close'].rolling(20).mean()
        minute_data['std20'] = minute_data['close'].rolling(20).std()
        minute_data['upper'] = minute_data['ma20'] + 2 * minute_data['std20']
        minute_data['lower'] = minute_data['ma20'] - 2 * minute_data['std20']

        for _, row in minute_data.iterrows():
            if pd.isna(row['ma20']):
                continue

            # 价格触及下轨 → 买入
            if row['close'] <= row['lower'] and position <= 0:
                qty = min(self.max_trade, self.holding)
                signals.append({
                    'time': row['time'],
                    'action': 'BUY',
                    'price': row['close'],
                    'quantity': qty
                })
                position += qty

            # 价格触及上轨 → 卖出底仓
            elif row['close'] >= row['upper'] and position >= 0:
                qty = min(self.max_trade, self.holding)
                signals.append({
                    'time': row['time'],
                    'action': 'SELL',
                    'price': row['close'],
                    'quantity': qty
                })
                position -= qty

        # 14:50 强制平仓（确保日内不遗留仓位）
        # （实际实现需检查时间和position）

        return signals
```

## 六、策略评估指标

```python
def evaluate_strategy(returns, benchmark_returns=None):
    """策略评估"""
    import numpy as np

    metrics = {}

    # 基本统计
    metrics['total_return'] = (1 + returns).prod() - 1
    metrics['annual_return'] = (1 + returns).prod() ** (252/len(returns)) - 1
    metrics['annual_volatility'] = returns.std() * np.sqrt(252)
    metrics['sharpe_ratio'] = metrics['annual_return'] / metrics['annual_volatility']

    # 最大回撤
    cummax = (1 + returns).cumprod().cummax()
    drawdown = (1 + returns).cumprod() / cummax - 1
    metrics['max_drawdown'] = drawdown.min()
    metrics['max_dd_duration'] = ...  # 最长回撤持续期

    # 胜率
    metrics['win_rate'] = (returns > 0).mean()
    metrics['avg_win'] = returns[returns > 0].mean()
    metrics['avg_loss'] = returns[returns < 0].mean()
    metrics['profit_factor'] = -returns[returns > 0].sum() / returns[returns < 0].sum()

    # 相对指标（如有基准）
    if benchmark_returns is not None:
        excess = returns - benchmark_returns
        metrics['alpha'] = excess.mean() * 252
        metrics['beta'] = returns.cov(benchmark_returns) / benchmark_returns.var()
        metrics['information_ratio'] = excess.mean() / excess.std() * np.sqrt(252)
        metrics['tracking_error'] = excess.std() * np.sqrt(252)

    return metrics
```
