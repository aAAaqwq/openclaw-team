## 二、特征工程

### 2.1 量化特征构造

```python
import pandas as pd
import numpy as np

class FeatureEngineer:
    """量化特征工程"""

    def __init__(self, df):
        """
        df: DataFrame with open, high, low, close, volume
        """
        self.df = df.copy()

    def price_features(self):
        """价格特征"""
        df = self.df

        # 收益率（多窗口）
        for n in [1, 3, 5, 10, 20, 60]:
            df[f'ret_{n}'] = df['close'].pct_change(n)

        # EMA偏离
        for n in [5, 10, 20, 60]:
            ema = df['close'].ewm(span=n).mean()
            df[f'ema{n}_bias'] = df['close'] / ema - 1

        # 价格位置（在N日高低之间的位置）
        for n in [10, 20, 60]:
            high = df['high'].rolling(n).max()
            low = df['low'].rolling(n).min()
            df[f'price_position_{n}'] = (df['close'] - low) / (high - low)

        return df

    def volume_features(self):
        """成交量特征"""
        df = self.df

        # 量比
        for n in [5, 10, 20]:
            df[f'vol_ratio_{n}'] = df['volume'] / df['volume'].rolling(n).mean()

        # 量价相关性
        for n in [10, 20]:
            df[f'price_vol_corr_{n}'] = (
                df['close'].pct_change().rolling(n)
                .corr(df['volume'].pct_change())
            )

        # OBV
        direction = np.sign(df['close'].diff())
        df['OBV'] = (direction * df['volume']).cumsum()
        df['OBV_MA'] = df['OBV'].rolling(20).mean()

        return df

    def technical_features(self):
        """技术指标特征"""
        df = self.df

        # RSI
        delta = df['close'].diff()
        gain = delta.clip(lower=0).rolling(14).mean()
        loss = (-delta).clip(lower=0).rolling(14).mean()
        df['RSI'] = 100 - 100 / (1 + gain / loss)

        # MACD
        ema12 = df['close'].ewm(span=12).mean()
        ema26 = df['close'].ewm(span=26).mean()
        df['MACD'] = ema12 - ema26
        df['MACD_signal'] = df['MACD'].ewm(span=9).mean()
        df['MACD_hist'] = df['MACD'] - df['MACD_signal']

        # ATR
        tr = pd.DataFrame({
            'hl': df['high'] - df['low'],
            'hc': (df['high'] - df['close'].shift(1)).abs(),
            'lc': (df['low'] - df['close'].shift(1)).abs()
        }).max(axis=1)
        df['ATR'] = tr.rolling(14).mean()
        df['ATR_pct'] = df['ATR'] / df['close']

        # BB宽度
        ma = df['close'].rolling(20).mean()
        std = df['close'].rolling(20).std()
        df['BB_width'] = 2 * std / ma
        df['BB_position'] = (df['close'] - ma + 2*std) / (4*std)

        return df

    def statistical_features(self):
        """统计特征"""
        df = self.df
        ret = df['close'].pct_change()

        # 偏度、峰度
        for n in [20, 60]:
            df[f'skew_{n}'] = ret.rolling(n).skew()
            df[f'kurt_{n}'] = ret.rolling(n).kurt()

        # 波动率
        for n in [5, 10, 20, 60]:
            df[f'vol_{n}'] = ret.rolling(n).std() * np.sqrt(252)

        # 自相关
        for n in [5, 10, 20]:
            df[f'autocorr_{n}'] = ret.rolling(n).apply(
                lambda x: pd.Series(x).autocorr(1) if len(x.dropna()) > 2 else 0
            )

        return df

    def build_all_features(self):
        """构建所有特征"""
        self.df = self.price_features()
        self.df = self.volume_features()
        self.df = self.technical_features()
        self.df = self.statistical_features()
        return self.df.dropna()
```

## 三、XGBoost 因子模型

### 3.1 训练流程

```python
# pip install xgboost scikit-learn
import xgboost as xgb
from sklearn.model_selection import TimeSeriesSplit
from sklearn.metrics import accuracy_score, classification_report
import numpy as np

class XGBoostFactorModel:
    """XGBoost因子预测模型"""

    def __init__(self, n_estimators=500, max_depth=6, learning_rate=0.05):
        self.model = xgb.XGBClassifier(
            n_estimators=n_estimators,
            max_depth=max_depth,
            learning_rate=learning_rate,
            subsample=0.8,
            colsample_bytree=0.8,
            reg_alpha=0.1,
            reg_lambda=1.0,
            random_state=42,
        )
        self.feature_cols = None

    def prepare_target(self, df, forward_period=5, quantile_threshold=0.3):
        """
        准备标签
        上30% → 1 (做多)
        下30% → 0 (做空)
        中间40% → 去掉（三分类问题简化为二分类）
        """
        forward_return = df['close'].pct_change(forward_period).shift(-forward_period)
        upper = forward_return.quantile(1 - quantile_threshold)
        lower = forward_return.quantile(quantile_threshold)

        df = df.copy()
        df['target'] = np.nan
        df.loc[forward_return >= upper, 'target'] = 1
        df.loc[forward_return <= lower, 'target'] = 0

        return df.dropna(subset=['target'])

    def train(self, df, feature_cols, target_col='target'):
        """训练模型"""
        self.feature_cols = feature_cols

        # 时间序列交叉验证
        tscv = TimeSeriesSplit(n_splits=5)
        cv_scores = []

        for train_idx, val_idx in tscv.split(df):
            X_train = df.iloc[train_idx][feature_cols]
            y_train = df.iloc[train_idx][target_col]
            X_val = df.iloc[val_idx][feature_cols]
            y_val = df.iloc[val_idx][target_col]

            self.model.fit(
                X_train, y_train,
                eval_set=[(X_val, y_val)],
                verbose=False,
            )

            pred = self.model.predict(X_val)
            acc = accuracy_score(y_val, pred)
            cv_scores.append(acc)

        print(f"CV Accuracy: {np.mean(cv_scores):.4f} ± {np.std(cv_scores):.4f}")

        # 全量训练
        self.model.fit(df[feature_cols], df[target_col])

        # 特征重要性
        importance = pd.Series(
            self.model.feature_importances_,
            index=feature_cols
        ).sort_values(ascending=False)

        print("\n特征重要性 Top 20:")
        print(importance.head(20))

        return self.model

    def predict(self, df):
        """预测"""
        proba = self.model.predict_proba(df[self.feature_cols])
        return pd.Series(proba[:, 1], index=df.index)  # 做多概率
```

### 3.2 使用示例

```python
# 完整流程
fe = FeatureEngineer(df)
data = fe.build_all_features()

# 特征列（排除价格和目标）
feature_cols = [c for c in data.columns
                if c not in ['open', 'high', 'low', 'close', 'volume', 'target']]

model = XGBoostFactorModel()
data = model.prepare_target(data, forward_period=5)
model.train(data, feature_cols)

# 预测
data['pred_proba'] = model.predict(data)
```

## 四、LSTM 时序模型

### 4.1 模型构建

```python
# pip install torch
import torch
import torch.nn as nn
import numpy as np

class LSTMPredictor(nn.Module):
    """LSTM价格预测模型"""

    def __init__(self, input_size, hidden_size=64, num_layers=2, dropout=0.2):
        super().__init__()
        self.hidden_size = hidden_size

        self.lstm = nn.LSTM(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout,
        )

        self.fc = nn.Sequential(
            nn.Linear(hidden_size, 32),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(32, 1),
        )

    def forward(self, x):
        # x: (batch, seq_len, features)
        lstm_out, _ = self.lstm(x)
        last_out = lstm_out[:, -1, :]  # 取最后一个时间步
        pred = self.fc(last_out)
        return pred


def create_sequences(data, seq_length=20):
    """创建时间序列样本"""
    X, y = [], []
    for i in range(len(data) - seq_length):
        X.append(data[i:i+seq_length])
        y.append(data[i+seq_length, 0])  # 假设第0列是目标
    return np.array(X), np.array(y)


# 训练
def train_lstm(X_train, y_train, input_size, epochs=100, lr=0.001):
    model = LSTMPredictor(input_size)
    criterion = nn.MSELoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=lr, weight_decay=1e-5)
    scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(
        optimizer, patience=10, factor=0.5
    )

    X_t = torch.FloatTensor(X_train)
    y_t = torch.FloatTensor(y_train).unsqueeze(1)

    for epoch in range(epochs):
        model.train()
        pred = model(X_t)
        loss = criterion(pred, y_t)

        optimizer.zero_grad()
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
        optimizer.step()
        scheduler.step(loss)

        if (epoch + 1) % 20 == 0:
            print(f"Epoch {epoch+1}, Loss: {loss.item():.6f}")

    return model
```

## 五、过拟合防范

### 5.1 过拟合检测

```python
class OverfitDetector:
    """过拟合检测工具"""

    @staticmethod
    def walk_forward_test(model_fn, df, feature_cols, target_col,
                          train_window=252, test_window=21):
        """
        Walk-Forward测试
        滚动训练+测试，更接近真实交易
        """
        results = []
        n = len(df)

        for start in range(0, n - train_window - test_window, test_window):
            train_end = start + train_window
            test_end = train_end + test_window

            train_data = df.iloc[start:train_end]
            test_data = df.iloc[train_end:test_end]

            # 训练
            model = model_fn()
            model.train(train_data, feature_cols, target_col)

            # 测试
            pred = model.predict(test_data)
            actual = test_data[target_col]

            # 计算IC
            from scipy.stats import spearmanr
            ic, _ = spearmanr(pred, actual)
            results.append(ic)

        mean_ic = np.mean(results)
        print(f"Walk-Forward IC: {mean_ic:.4f}")
        print(f"IC稳定性(std): {np.std(results):.4f}")
        print(f"IC胜率: {np.mean(np.array(results) > 0):.2%}")

        return results

    @staticmethod
    def purged_cross_validation(df, target_col, n_splits=5, embargo_pct=0.01):
        """
        带清洗和禁令期的交叉验证
        避免信息泄露
        """
        n = len(df)
        embargo = int(n * embargo_pct)
        fold_size = n // (n_splits + 1)

        for i in range(n_splits):
            test_start = (i + 1) * fold_size
            test_end = test_start + fold_size

            # 训练集：排除测试集前后embargo期
            train = pd.concat([
                df.iloc[:test_start - embargo],
                df.iloc[test_end + embargo:]
            ])
            test = df.iloc[test_start:test_end]

            yield train, test
```

### 5.2 防过拟合策略

| 策略 | 说明 | 优先级 |
|------|------|--------|
| **Walk-Forward** | 滚动训练测试 | ⭐⭐⭐⭐⭐ |
| **Purged CV** | 带禁令期的交叉验证 | ⭐⭐⭐⭐⭐ |
| **正则化** | L1/L2/Dropout | ⭐⭐⭐⭐ |
| **特征筛选** | 减少噪声特征 | ⭐⭐⭐⭐ |
| **样本外验证** | 至少留30%做样本外 | ⭐⭐⭐⭐⭐ |
| **简单模型优先** | 先用线性模型baseline | ⭐⭐⭐⭐⭐ |
| **组合优化** | 多模型投票 | ⭐⭐⭐ |
| **特征重要性** | 分析模型是否合理 | ⭐⭐⭐⭐ |

## 六、实战建议

1. **先跑通线性模型**：用Logistic回归做baseline，ML模型必须显著超过它才值得用
2. **Walk-Forward是标配**：不做Walk-Forward的ML回测结果不可信
3. **IC比准确率重要**：分类准确率可能因为类别不平衡而虚高
4. **组合优于单模型**：多个弱模型集成 > 一个强模型
5. **定期重训练**：ML模型的有效期有限，建议每1-3个月重训练
