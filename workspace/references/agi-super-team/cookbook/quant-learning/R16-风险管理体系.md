## 二、风险度量指标

### 2.1 VaR (Value at Risk)

```python
import numpy as np
import pandas as pd
from scipy import stats

def calc_var(returns, confidence=0.95, method='historical'):
    """
    计算VaR
    returns: 收益率序列
    confidence: 置信度 (0.95 或 0.99)
    method: 'historical', 'parametric', 'monte_carlo'
    """
    if method == 'historical':
        # 历史模拟法
        var = np.percentile(returns, (1 - confidence) * 100)

    elif method == 'parametric':
        # 参数法（假设正态分布）
        mu = returns.mean()
        sigma = returns.std()
        var = stats.norm.ppf(1 - confidence, mu, sigma)

    elif method == 'monte_carlo':
        # 蒙特卡洛模拟
        mu = returns.mean()
        sigma = returns.std()
        n_simulations = 100000
        simulated = np.random.normal(mu, sigma, n_simulations)
        var = np.percentile(simulated, (1 - confidence) * 100)

    return var

# 使用
daily_returns = df['close'].pct_change().dropna()
var_95 = calc_var(daily_returns, 0.95)
var_99 = calc_var(daily_returns, 0.99)

print(f"VaR(95%): {var_95:.2%}")  # 如: -2.5%，意味着95%概率日亏损不超过2.5%
print(f"VaR(99%): {var_99:.2%}")  # 如: -4.2%，意味着99%概率日亏损不超过4.2%
```

### 2.2 CVaR (Conditional VaR / Expected Shortfall)

```python
def calc_cvar(returns, confidence=0.95):
    """
    计算CVaR (条件VaR / 期望损失)
    CVaR = 超过VaR的损失的均值
    """
    var = calc_var(returns, confidence)
    cvar = returns[returns <= var].mean()
    return cvar

# CVaR比VaR更好地衡量尾部风险
print(f"CVaR(95%): {calc_cvar(daily_returns, 0.95):.2%}")
```

### 2.3 最大回撤

```python
def calc_max_drawdown(equity_curve):
    """
    计算最大回撤
    equity_curve: 净值序列
    """
    cummax = equity_curve.cummax()
    drawdown = equity_curve / cummax - 1
    max_dd = drawdown.min()
    max_dd_duration = (drawdown == max_dd).idxmax() - (equity_curve == cummax.loc[(drawdown == max_dd).idxmax()]).idxmax()

    return {
        'max_drawdown': max_dd,
        'max_dd_date': drawdown.idxmin(),
        'current_drawdown': drawdown.iloc[-1],
    }

def calc_calmar_ratio(annual_return, max_drawdown):
    """Calmar比率 = 年化收益 / 最大回撤"""
    return annual_return / abs(max_drawdown)
```

## 三、凯利公式与仓位管理

### 3.1 凯利公式

$$f^* = \frac{p \cdot b - q}{b}$$

其中：
- $f^*$：最优仓位比例
- $p$：胜率
- $q = 1 - p$：败率
- $b$：盈亏比（平均盈利/平均亏损）

```python
def kelly_criterion(win_rate, avg_win, avg_loss):
    """
    凯利公式计算最优仓位
    win_rate: 胜率
    avg_win: 平均盈利（正数）
    avg_loss: 平均亏损（正数）
    """
    b = avg_win / avg_loss  # 盈亏比
    q = 1 - win_rate
    kelly = (win_rate * b - q) / b

    # 实际使用中，通常用半凯利或1/3凯利
    return {
        'full_kelly': kelly,
        'half_kelly': kelly / 2,
        'third_kelly': kelly / 3,
        'recommended': kelly / 3,  # 推荐用1/3凯利
    }

# 示例
result = kelly_criterion(win_rate=0.55, avg_win=0.03, avg_loss=0.02)
print(f"最优仓位: {result['full_kelly']:.2%}")
print(f"推荐仓位(1/3凯利): {result['recommended']:.2%}")
```

### 3.2 反波动率仓位

```python
def inverse_volatility_weights(assets_returns):
    """
    反波动率加权
    波动率越低，分配越多
    """
    vols = assets_returns.std()
    inv_vols = 1 / vols
    weights = inv_vols / inv_vols.sum()
    return weights

def risk_parity_weights(assets_returns):
    """
    风险平价
    每个资产对组合风险的贡献相等
    """
    vols = assets_returns.std()
    corr = assets_returns.corr()

    # 简化版：用反波动率近似
    inv_vols = 1 / vols
    weights = inv_vols / inv_vols.sum()
    return weights
```

### 3.3 动态仓位管理

```python
class PositionManager:
    """动态仓位管理器"""

    def __init__(self, total_capital, max_position_pct=0.95,
                 max_single_pct=0.1, max_drawdown_pct=0.15):
        self.total_capital = total_capital
        self.max_position = max_position_pct
        self.max_single = max_single_pct
        self.max_drawdown = max_drawdown_pct
        self.peak_capital = total_capital

    def calculate_position(self, signal_strength, current_drawdown):
        """
        计算仓位大小
        signal_strength: 0-1的信号强度
        current_drawdown: 当前回撤（负数）
        """
        # 回撤调整：回撤越大，仓位越小
        dd_adjust = max(0, 1 + current_drawdown / self.max_drawdown)

        # 仓位 = 基础仓位 × 信号强度 × 回撤调整
        position = self.total_capital * self.max_position * signal_strength * dd_adjust

        return max(0, position)

    def update_capital(self, current_value):
        """更新资金"""
        self.total_capital = current_value
        self.peak_capital = max(self.peak_capital, current_value)

    def check_drawdown(self, current_value):
        """检查回撤"""
        dd = (current_value - self.peak_capital) / self.peak_capital
        if dd < -self.max_drawdown:
            return True  # 触发止损
        return False
```

## 四、止损策略

### 4.1 止损类型

| 止损类型 | 说明 | 适合 |
|----------|------|------|
| **固定止损** | 固定百分比（如5%） | 入门 |
| **ATR止损** | N倍ATR | 趋势策略 |
| **追踪止损** | 从最高点回落X% | 趋势策略 |
| **时间止损** | 持有N天未达目标退出 | 日内/短线 |
| **信号止损** | 信号反转即退出 | 因子策略 |
| **波动率止损** | 波动率突变退出 | 高波动环境 |

### 4.2 综合止损系统

```python
class StopLossSystem:
    """综合止损系统"""

    def __init__(self, initial_capital):
        self.capital = initial_capital
        self.peak = initial_capital

    def check_all_stops(self, entry_price, current_price, holding_days,
                        current_atr, signal_value):
        """
        检查所有止损条件
        返回: (should_stop, reason)
        """
        pnl_pct = (current_price - entry_price) / entry_price

        # 1. 固定止损 (-5%)
        if pnl_pct < -0.05:
            return True, f"固定止损: {pnl_pct:.2%}"

        # 2. ATR止损 (2倍ATR)
        if current_price < entry_price - 2 * current_atr:
            return True, f"ATR止损: 价格低于入场价-2ATR"

        # 3. 时间止损 (持有10天，收益<2%)
        if holding_days >= 10 and pnl_pct < 0.02:
            return True, f"时间止损: 持有{holding_days}天，收益仅{pnl_pct:.2%}"

        # 4. 信号止损
        if signal_value < 0:  # 信号反转
            return True, "信号止损: 信号反转"

        # 5. 组合级止损
        current_value = self.capital * (1 + pnl_pct)
        self.peak = max(self.peak, current_value)
        dd = (current_value - self.peak) / self.peak
        if dd < -0.10:  # 10%组合止损
            return True, f"组合止损: 回撤{dd:.2%}"

        return False, ""
```

## 五、风险预算

### 5.1 组合风险分配

```python
def risk_budgeting(cov_matrix, risk_budget):
    """
    风险预算分配
    cov_matrix: 协方差矩阵
    risk_budget: 各资产的风险预算比例
    """
    n = len(risk_budget)
    weights = np.ones(n) / n  # 初始等权

    for _ in range(1000):  # 迭代优化
        # 各资产的风险贡献
        portfolio_vol = np.sqrt(weights @ cov_matrix @ weights)
        marginal_risk = cov_matrix @ weights / portfolio_vol
        risk_contribution = weights * marginal_risk
        risk_contribution_pct = risk_contribution / risk_contribution.sum()

        # 调整权重
        adjustment = risk_budget / risk_contribution_pct
        weights = weights * adjustment
        weights = weights / weights.sum()

    return weights
```
